generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mysql"
  url      = env("DATABASE_URL")
}

enum Role {
  farmer
  buyer
}

enum OrderStatus {
  pending
  confirmed
  shipped
  delivered
}

enum OrderSource {
  cart
  quotation
}

enum QuotationStatus {
  pending
  accepted
  rejected
}

enum ProductUnit {
  kg
  pcs
}

model User {
  id        Int       @id @default(autoincrement())
  name      String
  email     String    @unique
  password  String
  role      Role
  phone     String?
  address   String?
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt

  products        Product[]
  carts           Cart[]          @relation("UserCarts")
  farmerCarts     Cart[]          @relation("FarmerCarts")
  quotations      Quotation[]      @relation("UserQuotations")
  farmerQuotes    Quotation[]      @relation("FarmerQuotations")
  orders          Order[]         @relation("UserOrders")
  farmerOrders    Order[]         @relation("FarmerOrders")

  @@index([email])
  @@index([role])
}

model Product {
  id               Int               @id @default(autoincrement())
  title            String
  description      String?
  nutritionalInfo  String?
  unit             ProductUnit       @default(kg)
  price            Decimal           @db.Decimal(10, 2)
  quantityAvailable Int               @default(0)
  isActive         Boolean           @default(true)
  isDeleted        Boolean           @default(false)
  createdAt        DateTime          @default(now())
  updatedAt        DateTime          @updatedAt
  farmerId         Int
  farmer           User              @relation(fields: [farmerId], references: [id])

  images           ProductImage[]
  categories       ProductCategory[]
  cartItems        CartItem[]
  quotationItems   QuotationItem[]
  orderItems       OrderItem[]

  @@index([farmerId])
  @@index([isActive])
}

model ProductImage {
  id        Int     @id @default(autoincrement())
  url       String
  productId Int
  product   Product @relation(fields: [productId], references: [id], onDelete: Cascade)

  @@index([productId])
}

model Category {
  id        Int               @id @default(autoincrement())
  name      String            @unique
  products  ProductCategory[]

  @@index([name])
}

model ProductCategory {
  productId  Int
  categoryId Int
  product    Product  @relation(fields: [productId, categoryId], references: [id], onDelete: Cascade)
  category   Category @relation(fields: [categoryId], references: [id], onDelete: Cascade)

  @@id([productId, categoryId])
}

model Cart {
  id        Int        @id @default(autoincrement())
  createdAt DateTime   @default(now())
  updatedAt DateTime   @updatedAt
  userId    Int        @unique
  farmerId  Int
  user      User       @relation("UserCarts", fields: [userId], references: [id])
  farmer    User       @relation("FarmerCarts", fields: [farmerId], references: [id])
  items     CartItem[]

  @@index([userId])
  @@index([farmerId])
}

model CartItem {
  id        Int     @id @default(autoincrement())
  quantity  Int
  cartId    Int
  productId Int
  cart      Cart    @relation(fields: [cartId], references: [id], onDelete: Cascade)
  product   Product @relation(fields: [productId], references: [id])

  @@unique([cartId, productId])
  @@index([cartId])
  @@index([productId])
}

model Quotation {
  id          Int             @id @default(autoincrement())
  status      QuotationStatus @default(pending)
  items       Json
  createdAt   DateTime        @default(now())
  updatedAt   DateTime        @updatedAt
  userId      Int
  farmerId    Int
  user        User            @relation("UserQuotations", fields: [userId], references: [id])
  farmer      User            @relation("FarmerQuotations", fields: [farmerId], references: [id])

  @@index([userId])
  @@index([farmerId])
  @@index([status])
}

model Order {
  id             Int          @id @default(autoincrement())
  totalAmount    Decimal      @db.Decimal(10, 2)
  status         OrderStatus  @default(pending)
  source         OrderSource
  paymentStatus  String       @default("pending")
  shippingAddress String?
  items          Json
  createdAt      DateTime     @default(now())
  updatedAt      DateTime     @updatedAt
  userId         Int
  farmerId       Int
  user           User         @relation("UserOrders", fields: [userId], references: [id])
  farmer         User         @relation("FarmerOrders", fields: [farmerId], references: [id])

  @@index([userId])
  @@index([farmerId])
  @@index([status])
}
